1.写一个函数，名字叫min，带两个参数，返回其中最小值

function min(a,b){
  return a<b?a:b;
}
console.log("min is "+min(5,"6"));

2.写一个函数，名字叫countBs，带一个字符串参数，返回这个字符串中有多少个大写的”B”


function countBs(s){
  var res = 0;
  for(var i = 0; i<s.length; i++) {
    if(s.charAt(i) === "B"){
      res++;
    }
  }
  return res;
}
console.log("contB is "+countBs("Bye bye holimo,Bob"));

3.写一个函数，名字叫countChar，比刚才的countBs函数多一个参数，这第二个参数用来指示算哪个字符出现的次数


function countChar(s,c){
  var res = 0;
  for(var i = 0; i<s.length; i++) {
    if(s.charAt(i) === c){
      res++;
    }
  }
  return res;
}
console.log("count char is "+countChar("Bye bye holimo,Bob","o"));

4.写一个函数，名字叫genCountChar，带一个参数指示要计算哪个字符，返回一个计算此字符出现次数的，针对字符串的计数函数，这个字符计数函数的工作方式跟countBs一样



function genCountChar(c){
  
  return function(s){
    var res = 0;
    for(var i = 0; i<s.length; i++) {
      if(s.charAt(i) === c){
        res++;
      }
    }
    return res;
  };
}

var foo = genCountChar("e");
console.log("count char is "+foo("Bye bye holimo,Bob"));
